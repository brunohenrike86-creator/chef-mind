// ============================================ // chef-mind · RestERP — Google Apps Script API // Versão: 1.0 | Fase 2 // ============================================ // Como usar: // No Google Sheets → Extensões → Apps Script // Cole este código, salve e publique como Web App // ============================================ // ---------- UTILITÁRIOS ---------- function getSheet(name) { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name); } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i]; }); return obj; }); } function gerarId() { return Date.now().toString(36) + Math.random().toString(36).substr(2, 5); } function agora() { return new Date().toISOString(); } function resposta(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ---------- LEITURAS (GET) ---------- function getCardapio() { const itens = sheetToObjects(getSheet('cardapio')); return itens.filter(i => String(i.ativo).toLowerCase() === 'true'); } function getConfig() { const rows = sheetToObjects(getSheet('config')); const config = {}; rows.forEach(r => { config[r.chave] = r.valor; }); return config; } function getUsuarios() { return sheetToObjects(getSheet('usuarios')) .filter(u => String(u.ativo).toLowerCase() === 'true'); } function _hojeStr() { return new Date().toISOString().split('T')[0]; } function _dataComanda(cmd) { return cmd.timestamp_abertura ? String(cmd.timestamp_abertura).split('T')[0] : null; } // Retorna apenas comandas ativas (aberta/em_preparo) de hoje, sem encerramento function getComandasAbertas() { const hoje = _hojeStr(); return sheetToObjects(getSheet('comandas')) .filter(c => (c.status === 'aberta' || c.status === 'em_preparo') && !c.encerramento && _dataComanda(c) === hoje ); } // Retorna TODAS as comandas de hoje que não foram encerradas (abertas + pagas) // Usada pelo polling para manter estado de comandas pagas no frontend function getComandasDoDia() { const hoje = _hojeStr(); return sheetToObjects(getSheet('comandas')) .filter(c => _dataComanda(c) === hoje && !c.encerramento); } function getTodasComandas() { return sheetToObjects(getSheet('comandas')); } function getComanda(id) { const comandas = sheetToObjects(getSheet('comandas')); return comandas.find(c => String(c.id) === String(id)) || null; } function getItensDaComanda(comanda_id) { return sheetToObjects(getSheet('itens')) .filter(i => String(i.comanda_id) === String(comanda_id)); } function getItensPreparo() { // Retorna itens com preparo_ja=true ainda não concluídos (para tela de preparo) return sheetToObjects(getSheet('itens')) .filter(i => String(i.preparo_ja).toLowerCase() === 'true' && !i.timestamp_concluido ); } // ---------- ESCRITAS (POST) ---------- function login(data) { const usuarios = sheetToObjects(getSheet('usuarios')); const user = usuarios.find(u => u.login === data.login && String(u.senha) === String(data.senha) && String(u.ativo).toLowerCase() === 'true' ); if (user) { return { sucesso: true, nivel: user.nivel, login: user.login }; } return { sucesso: false, erro: 'Login ou senha incorretos' }; } function novaComanda(data) { const sheet = getSheet('comandas'); const id = gerarId(); sheet.appendRow([ id, // id data.numero, // numero 'aberta', // status agora(), // timestamp_abertura '', // timestamp_pagamento 0, // total '', // forma_pagamento false // preparo_ja ]); // Campos extras (auto-criam a coluna se não existir) if (data.usuario) _setCampo(sheet, 'id', id, 'usuario_abertura', data.usuario); return { sucesso: true, id }; } function addItem(data) { const sheet = getSheet('itens'); const id = gerarId(); const total = (data.quantidade || 1) * (data.preco_unit || 0); sheet.appendRow([ id, // id data.comanda_id, // comanda_id data.codigo, // codigo data.nome, // nome data.categoria || '', // categoria data.quantidade || 1, // quantidade data.preco_unit || 0, // preco_unit total, // total data.observacao || '', // observacao data.preparo_ja || false, // preparo_ja agora(), // timestamp_pedido '', // timestamp_inicio_preparo '', // timestamp_concluido '' // usuario_preparo ]); // Campo usuário (auto-cria coluna se não existir) if (data.usuario) _setCampo(getSheet('itens'), 'id', id, 'usuario_lancamento', data.usuario); // Recalcular e gravar total da comanda _recalcularTotalComanda(data.comanda_id); // Se Preparo p/ mesa, atualiza status da comanda para em_preparo if (data.preparo_ja) { _setStatusComanda(data.comanda_id, 'em_preparo'); _setPreparoJaComanda(data.comanda_id); } return { sucesso: true, id }; } function iniciarPreparo(data) { // Chamado quando atendente clica "INICIAR" na tela de preparo const sheet = getSheet('itens'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); const inicioCol = headers.indexOf('timestamp_inicio_preparo'); const usuarioCol = headers.indexOf('usuario_preparo'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(data.item_id)) { sheet.getRange(i + 1, inicioCol + 1).setValue(agora()); sheet.getRange(i + 1, usuarioCol + 1).setValue(data.usuario || ''); return { sucesso: true }; } } return { erro: 'Item não encontrado' }; } function concluirPreparo(data) { // Chamado quando atendente clica "CONCLUÍDO" na tela de preparo const sheet = getSheet('itens'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); const concluidoCol = headers.indexOf('timestamp_concluido'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(data.item_id)) { sheet.getRange(i + 1, concluidoCol + 1).setValue(agora()); return { sucesso: true }; } } return { erro: 'Item não encontrado' }; } function pagarComanda(data) { const sheet = getSheet('comandas'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); const statusCol = headers.indexOf('status'); const pagamentoCol = headers.indexOf('timestamp_pagamento'); const totalCol = headers.indexOf('total'); const formaPagCol = headers.indexOf('forma_pagamento'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(data.comanda_id)) { sheet.getRange(i + 1, statusCol + 1).setValue('paga'); sheet.getRange(i + 1, pagamentoCol + 1).setValue(agora()); sheet.getRange(i + 1, totalCol + 1).setValue(data.total || 0); sheet.getRange(i + 1, formaPagCol + 1).setValue(data.forma_pagamento || ''); if (data.usuario) _setCampo(sheet, 'id', data.comanda_id, 'usuario_pagamento', data.usuario); return { sucesso: true }; } } return { erro: 'Comanda não encontrada' }; } function atualizarStatusComanda(data) { _setStatusComanda(data.id, data.status); return { sucesso: true }; } // ---------- GESTÃO DE USUÁRIOS ---------- function novoUsuario(data) { const sheet = getSheet('usuarios'); // Verificar se login já existe const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const loginCol = headers.indexOf('login'); for (let i = 1; i < rows.length; i++) { if (rows[i][loginCol] === data.login) { return { erro: 'Usuário já existe' }; } } sheet.appendRow([data.login, data.senha, data.nivel || 'atendente', true]); return { sucesso: true }; } function atualizarUsuario(data) { const sheet = getSheet('usuarios'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const loginCol = headers.indexOf('login'); const senhaCol = headers.indexOf('senha'); const nivelCol = headers.indexOf('nivel'); for (let i = 1; i < rows.length; i++) { if (rows[i][loginCol] === data.login) { if (data.senha !== undefined) sheet.getRange(i + 1, senhaCol + 1).setValue(data.senha); if (data.nivel !== undefined) sheet.getRange(i + 1, nivelCol + 1).setValue(data.nivel); return { sucesso: true }; } } return { erro: 'Usuário não encontrado' }; } function deletarUsuario(data) { const sheet = getSheet('usuarios'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const loginCol = headers.indexOf('login'); const ativoCol = headers.indexOf('ativo'); for (let i = 1; i < rows.length; i++) { if (rows[i][loginCol] === data.login) { sheet.getRange(i + 1, ativoCol + 1).setValue(false); return { sucesso: true }; } } return { erro: 'Usuário não encontrado' }; } function resetarSenha(data) { return atualizarUsuario({ login: data.login, senha: data.senha }); } function removerItem(data) { // Marca o item como removido na aba itens const sheet = getSheet('itens'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); // Adiciona coluna 'removido' se não existir let removidoCol = headers.indexOf('removido'); if (removidoCol === -1) { const lastCol = headers.length + 1; sheet.getRange(1, lastCol).setValue('removido'); removidoCol = lastCol - 1; } for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(data.item_id)) { sheet.getRange(i + 1, removidoCol + 1).setValue(true); // Recalcular total da comanda após remoção const comanda_id = rows[i][headers.indexOf('comanda_id')]; if (comanda_id) _recalcularTotalComanda(comanda_id); return { sucesso: true }; } } return { erro: 'Item não encontrado' }; } function atualizarCardapio(data) { const sheet = getSheet('cardapio'); if (data.operacao === 'novo') { sheet.appendRow([data.codigo, data.nome, data.categoria, data.preco, true]); return { sucesso: true }; } if (data.operacao === 'atualizar' || data.operacao === 'inativar') { const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const codigoCol = headers.indexOf('codigo'); const precoCol = headers.indexOf('preco'); const ativoCol = headers.indexOf('ativo'); const nomeCol = headers.indexOf('nome'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][codigoCol]) === String(data.codigo)) { if (data.preco !== undefined) sheet.getRange(i + 1, precoCol + 1).setValue(data.preco); if (data.ativo !== undefined) sheet.getRange(i + 1, ativoCol + 1).setValue(data.ativo); if (data.nome !== undefined) sheet.getRange(i + 1, nomeCol + 1).setValue(data.nome); return { sucesso: true }; } } return { erro: 'Item do cardápio não encontrado: ' + data.codigo }; } return { erro: 'Operação inválida. Use: novo | atualizar | inativar' }; } function atualizarConfig(data) { // data = { chave: "carnes_seg", valor: "Frango,Carne,Peixe" } const sheet = getSheet('config'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const chaveCol = headers.indexOf('chave'); const valorCol = headers.indexOf('valor'); for (let i = 1; i < rows.length; i++) { if (rows[i][chaveCol] === data.chave) { sheet.getRange(i + 1, valorCol + 1).setValue(data.valor); return { sucesso: true }; } } // Chave nova — adiciona sheet.appendRow([data.chave, data.valor]); return { sucesso: true }; } // ---------- RESET E FECHAMENTO ---------- function _setEncerramentoComanda(id, tipo) { const sheet = getSheet('comandas'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); // Auto-cria coluna encerramento se não existir let encCol = headers.indexOf('encerramento'); if (encCol === -1) { encCol = headers.length; sheet.getRange(1, encCol + 1).setValue('encerramento'); } for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(id)) { sheet.getRange(i + 1, encCol + 1).setValue(tipo); return true; } } return false; } // Encerra urgência: marca comandas abertas/em_preparo de hoje como encerramento='reset' // NÃO altera status, NÃO apaga dados históricos function resetarComandas() { const hoje = _hojeStr(); const comandas = sheetToObjects(getSheet('comandas')); let count = 0; comandas.forEach(c => { if ( _dataComanda(c) === hoje && (c.status === 'aberta' || c.status === 'em_preparo') && !c.paid && !c.encerramento ) { _setEncerramentoComanda(c.id, 'reset'); count++; } }); return { sucesso: true, encerradas: count }; } // Fecha expediente: marca comandas abertas não pagas de hoje como encerramento='fechamento' // NÃO altera status, NÃO apaga dados históricos function fecharExpediente() { const hoje = _hojeStr(); const comandas = sheetToObjects(getSheet('comandas')); let count = 0; comandas.forEach(c => { if ( _dataComanda(c) === hoje && (c.status === 'aberta' || c.status === 'em_preparo') && !c.paid && !c.encerramento ) { _setEncerramentoComanda(c.id, 'fechamento'); count++; } }); return { sucesso: true, encerradas: count }; } // ---------- FUNÇÕES INTERNAS ---------- // Grava qualquer campo em qualquer sheet usando headers (auto-cria coluna se não existir) function _setCampo(sheet, idColName, idVal, campo, valor) { const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf(idColName); let campoCol = headers.indexOf(campo); if (campoCol === -1) { campoCol = headers.length; sheet.getRange(1, campoCol + 1).setValue(campo); } for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(idVal)) { sheet.getRange(i + 1, campoCol + 1).setValue(valor); return true; } } return false; } // Recalcula e grava o total da comanda somando itens não removidos function _recalcularTotalComanda(comanda_id) { const itens = sheetToObjects(getSheet('itens')) .filter(i => String(i.comanda_id) === String(comanda_id) && String(i.removido).toLowerCase() !== 'true' ); const total = itens.reduce((sum, i) => sum + (parseFloat(i.total) || 0), 0); _setCampo(getSheet('comandas'), 'id', comanda_id, 'total', total); return total; } function _setStatusComanda(id, status) { const sheet = getSheet('comandas'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); const statusCol = headers.indexOf('status'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(id)) { sheet.getRange(i + 1, statusCol + 1).setValue(status); return; } } } function _setPreparoJaComanda(id) { const sheet = getSheet('comandas'); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const idCol = headers.indexOf('id'); const preparoCol = headers.indexOf('preparo_ja'); for (let i = 1; i < rows.length; i++) { if (String(rows[i][idCol]) === String(id)) { sheet.getRange(i + 1, preparoCol + 1).setValue(true); return; } } } // ---------- ROTEADOR GET ---------- function doGet(e) { const action = e.parameter.action; let result; try { switch (action) { case 'getCardapio': result = getCardapio(); break; case 'getConfig': result = getConfig(); break; case 'getUsuarios': result = getUsuarios(); break; case 'getComandasAbertas': result = getComandasAbertas(); break; case 'getComandasDoDia': result = getComandasDoDia(); break; case 'getTodasComandas': result = getTodasComandas(); break; case 'getComanda': result = getComanda(e.parameter.id); break; case 'getItensDaComanda': result = getItensDaComanda(e.parameter.comanda_id); break; case 'getItensPreparo': result = getItensPreparo(); break; default: result = { erro: 'Ação GET desconhecida: ' + action }; } } catch (err) { result = { erro: err.message }; } return resposta(result); } // ---------- ROTEADOR POST ---------- function doPost(e) { let data; try { data = JSON.parse(e.postData.contents); } catch (err) { return resposta({ erro: 'JSON inválido: ' + err.message }); } const action = data.action; let result; try { switch (action) { case 'login': result = login(data); break; case 'novaComanda': result = novaComanda(data); break; case 'addItem': result = addItem(data); break; case 'iniciarPreparo': result = iniciarPreparo(data); break; case 'concluirPreparo': result = concluirPreparo(data); break; case 'pagarComanda': result = pagarComanda(data); break; case 'atualizarStatusComanda': result = atualizarStatusComanda(data); break; case 'novoUsuario': result = novoUsuario(data); break; case 'atualizarUsuario': result = atualizarUsuario(data); break; case 'deletarUsuario': result = deletarUsuario(data); break; case 'resetarSenha': result = resetarSenha(data); break; case 'removerItem': result = removerItem(data); break; case 'atualizarCardapio': result = atualizarCardapio(data); break; case 'atualizarConfig': result = atualizarConfig(data); break; case 'resetarComandas': result = resetarComandas(); break; case 'fecharExpediente': result = fecharExpediente(); break; default: result = { erro: 'Ação POST desconhecida: ' + action }; } } catch (err) { result = { erro: err.message }; } return resposta(result); }
